function calculatePositionSize(atr, accountBalance) {
  const riskPerTrade = accountBalance * 0.01; // 1%风险
  const stopLossPips = atr * 1.5;
  const pipValue = 0.01; // USDJPY 1pip ≈ $0.1 per 0.01 lot
  const lotSize = riskPerTrade / (stopLossPips * pipValue * 100);
  return Math.min(lotSize, 0.5); // 最大0.5手
}
